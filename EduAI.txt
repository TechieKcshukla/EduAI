    {text:`A student claims: "${topics[1]||topics[0]} always follows the basic rule." Is this true?`, opts:["Yes — the rule always holds","No — exceptions exist in all cases","Only under specific conditions","The claim is irrelevant"], ci:0},
    {text:`Apply the concept of ${topics[0]||"the main topic"} from "${chapterName}" to solve a standard problem. Which approach is valid?`, opts:["Use direct formula application","Ignore boundary conditions","Apply inverse method only","Random trial and error"], ci:0},

    {text:`Critically evaluate: how does "${topics[0]||"the core concept"}" interconnect with other ideas in "${chapterName}"?`, opts:["Multiple deep connections exist and are well-defined","No meaningful connections exist","Only superficial links are present","The concepts are entirely independent"], ci:0},

  const errG  = document.getElementById('li-err-general');
  errG.style.display = 'none';

  if (!email.includes('@')) { showErr('li-err-email','Enter a valid email'); return; }
  hideErr('li-err-email');
  if (!pass) { showErr('li-err-pass','Password is required'); return; }
  hideErr('li-err-pass');

  const user = USERS_DB[email];

    errG.textContent = '❌ No account found with this email. Please register first.';

  }

  // Login success — launch app
  launchApp({ name:user.name, email:email, avatar:user.avatar, board:user.board, cls:user.cls });
}

// Password strength indicator
document.addEventListener('DOMContentLoaded', () => {
  const passInput = document.getElementById('reg-pass');
  if (passInput) {
    passInput.addEventListener('input', () => {
      const v = passInput.value;
      const bar = document.getElementById('pass-strength-bar');
      const lbl = document.getElementById('pass-strength-label');
      if (!v) { bar.style.width='0%'; lbl.textContent=''; return; }
      let score = 0;
      if (v.length >= 6) score++;
      if (v.length >= 10) score++;
      if (/[A-Z]/.test(v)) score++;
      if (/[0-9]/.test(v)) score++;
      if (/[^A-Za-z0-9]/.test(v)) score++;
      const configs = [
        {w:'20%', c:'#ff6b6b', l:'Weak'},
        {w:'40%', c:'#ff6b6b', l:'Weak'},
        {w:'60%', c:'#ffb930', l:'Fair'},
        {w:'80%', c:'#ffb930', l:'Good'},
        {w:'100%',c:'#00e5a0', l:'Strong'},
      ];
      const cfg = configs[Math.min(score, configs.length-1)];
      bar.style.width = cfg.w; bar.style.background = cfg.c;
      lbl.textContent = cfg.l; lbl.style.color = cfg.c;
    });
  }
});

function makeRadial(value, size, color, label) {
  const r = (size - 10) / 2;
  const circ = 2 * Math.PI * r;
  const offset = circ - (Math.max(0, Math.min(100, value)) / 100) * circ;
  return `<div class="radial-wrap">
    <svg width="${size}" height="${size}" style="overflow:visible">
      <circle cx="${size/2}" cy="${size/2}" r="${r}" fill="none" stroke="#2c3148" stroke-width="6"/>
      <circle cx="${size/2}" cy="${size/2}" r="${r}" fill="none" stroke="${color}" stroke-width="6"
        stroke-dasharray="${circ}" stroke-dashoffset="${offset}" stroke-linecap="round"
        style="transform:rotate(-90deg);transform-origin:center;transition:stroke-dashoffset 1.1s ease"/>
      <text x="50%" y="50%" text-anchor="middle" dy=".35em" fill="${color}"
        font-size="${size<80?12:15}" font-weight="700" font-family="Syne,sans-serif">${value}%</text>
    </svg>
    ${label ? `<div class="radial-label">${label}</div>` : ''}
  </div>`;
}

function makeBar(value, color, label, right, h=6) {
  return `<div class="bar-wrap">
    ${(label||right) ? `<div class="bar-meta">${label?`<span class="bar-label">${label}</span>`:''} ${right?`<span class="bar-val">${right}</span>`:''}</div>` : ''}
    <div class="bar-track" style="height:${h}px">
      <div class="bar-fill" style="width:${Math.max(0,Math.min(100,value))}%;background:${color};height:${h}px"></div>
    </div>
  </div>`;
}

function makeBadge(text, color) {
  return `<span class="badge" style="background:${color}18;color:${color};border:1px solid ${color}30">${text}</span>`;
}

function makeTag(text, color) {
  return `<span class="tag" style="background:${color}15;color:${color};border:1px solid ${color}28">${text}</span>`;
}

function avg(arr, key) {
  if (!arr.length) return 0;
  return Math.round(arr.reduce((s,r) => s + (r.metrics ? r.metrics[key] : r[key]), 0) / arr.length);
}

function getSubjColor(subj) {
  return activeSyllabus[subj]?.color || '#00e5a0';
}

function getAllChapters() {
  return Object.keys(activeSyllabus).flatMap(s => activeSyllabus[s].chapters.map(c => ({...c, subject:s})));
}

function renderDashboard() {
  const subjects = Object.keys(activeSyllabus);
  const allChs = getAllChapters();
  const n = allResults.length;
  const doneCh = Object.values(chapterStats).filter(s => s.completed).length;
  const avgAcc = avg(allResults, 'accuracy');
  const avgSpd = avg(allResults, 'speedScore');
  const passRate = n > 0 ? Math.round((allResults.filter(r => r.metrics.passed).length / n) * 100) : 0;

  document.getElementById('dash-greeting').textContent = `Welcome back, ${student.name.split(' ')[0]} ${student.avatar}`;
  document.getElementById('dash-date').textContent = new Date().toLocaleDateString('en-IN', {weekday:'long',month:'long',day:'numeric',year:'numeric'});

  // KPIs
  const kpis = [
    {l:'Chapters Done', v:`${doneCh}`, sub:`of ${allChs.length}`, icon:'✅', c:'#00e5a0'},
    {l:'Avg Accuracy', v:`${avgAcc}%`, sub:'all tests', icon:'🎯', c:'#ffb930'},
    {l:'Speed Index', v:`${avgSpd}%`, sub:'vs expected', icon:'⚡', c:'#38bdf8'},
    {l:'Pass Rate', v:`${passRate}%`, sub:`${allResults.filter(r=>r.metrics.passed).length}/${n} tests`, icon:'🏆', c:'#a78bfa'}
  ];
  document.getElementById('kpi-row').innerHTML = kpis.map((k,i) => `
    <div class="kpi-card" style="animation-delay:${i*.08}s;animation:fadeUp .5s ease both">
      <div class="kpi-header">
        <span class="kpi-icon">${k.icon}</span>
        <div class="kpi-pulse" style="background:${k.c};box-shadow:0 0 8px ${k.c}"></div>
      </div>
      <div class="kpi-value" style="color:${k.c}">${k.v}</div>
      <div class="kpi-label">${k.l}</div>
      <div class="kpi-sub">${k.sub}</div>
    </div>`).join('');

  // Subject progress
  document.getElementById('subj-progress-list').innerHTML = subjects.map(s => {
    const chs = activeSyllabus[s].chapters;
    const done = chs.filter(c => chapterStats[c.id]?.completed).length;
    const pct = chs.length > 0 ? Math.round((done/chs.length)*100) : 0;
    const color = activeSyllabus[s].color || '#00e5a0';
    return `<div class="subj-prog-item">
      <div class="subj-prog-header">
        <div class="subj-prog-name"><span>${activeSyllabus[s].icon}</span>${s.replace('_',' ')}</div>
        ${makeTag(`${done}/${chs.length}`, color)}
      </div>
      ${makeBar(pct, color, '', '', 7)}
    </div>`;
  }).join('');

  // Performance overview
  if (n === 0) {
    document.getElementById('perf-overview').innerHTML = `<div class="text-center" style="padding:24px">
      <div style="font-size:32px;margin-bottom:8px">📊</div>
      <p style="color:var(--t2);font-size:13px;margin-bottom:12px">Complete a test to see breakdown</p>
      <button class="btn btn-secondary btn-sm" onclick="showPage('practice', document.querySelectorAll('.nav-item')[2])">Start Practice</button>
    </div>`;
  } else {
    const easyA = avg(allResults, 'easyAcc'), medA = avg(allResults, 'medAcc'), hardA = avg(allResults, 'hardAcc');
    const perf = Math.round(avgAcc*.6 + avgSpd*.4);
    document.getElementById('perf-overview').innerHTML = `
      <div class="radial-row">${makeRadial(avgAcc,78,'#00e5a0','Accuracy')}${makeRadial(avgSpd,78,'#ffb930','Speed')}${makeRadial(perf,78,'#a78bfa','Overall')}</div>
      ${makeBar(easyA,'#00e5a0','Easy',`${easyA}%`,5)}
      ${makeBar(medA,'#ffb930','Medium',`${medA}%`,5)}
      ${makeBar(hardA,'#ff6b6b','Hard',`${hardA}%`,5)}`;
  }

  // Weak chapters
  const weak = getAllChapters().filter(c => { const s=chapterStats[c.id]; return s?.attempts>0&&!s.completed; })
    .map(c=>({...c,score:chapterStats[c.id]?.bestScore||0})).sort((a,b)=>a.score-b.score).slice(0,3);
  document.getElementById('weak-list').innerHTML = weak.length === 0
    ? `<p style="color:var(--t2);font-size:13px">${n===0?'Start practicing':'No weak chapters!'}</p>`
    : weak.map(c => `<div class="activity-item" style="cursor:pointer" onclick="goToPractice('${c.subject}','${c.id}')">
      <div><div class="act-title">${c.name}</div><div class="act-date">${c.subject?.replace('_',' ')}</div></div>
      ${makeBadge(c.score+'%', c.score<40?'#ff6b6b':'#ffb930')}
    </div>`).join('');

  // Strong chapters
  const strong = getAllChapters().filter(c => chapterStats[c.id]?.completed)
    .map(c=>({...c,score:chapterStats[c.id]?.bestScore||0})).sort((a,b)=>b.score-a.score).slice(0,3);
  document.getElementById('strong-list').innerHTML = strong.length === 0
    ? `<p style="color:var(--t2);font-size:13px">Complete chapters to see strengths</p>`
    : strong.map(c => `<div class="activity-item">
      <div><div class="act-title">${c.name}</div><div class="act-date">${c.subject?.replace('_',' ')}</div></div>
      ${makeBadge(c.score+'%','#00e5a0')}
    </div>`).join('');

  // Recent tests
  const recent = [...allResults].sort((a,b) => b.timestamp-a.timestamp).slice(0,4);
  document.getElementById('recent-list').innerHTML = recent.length === 0
    ? `<p style="color:var(--t2);font-size:13px">No tests yet</p>`
    : recent.map(r => `<div class="activity-item">
      <div><div class="act-title">${r.chapterName}</div><div class="act-date">${new Date(r.timestamp).toLocaleDateString()}</div></div>
      <div><div class="act-score" style="color:${r.metrics.passed?'#00e5a0':'#ff6b6b'}">${r.metrics.accuracy}%</div>
      <div class="act-status" style="color:${r.metrics.passed?'#00e5a0':'#ff6b6b'}">${r.metrics.passed?'✓ Pass':'✗ Fail'}</div></div>
    </div>`).join('');
}

function renderQuestion() {
  const q = pracQuestions[pracQIdx];
  if (!q) return;
  const dc = {Easy:'#00e5a0', Medium:'#ffb930', Hard:'#ff6b6b'}[q.difficulty];
  const prog = ((pracQIdx+1)/pracQuestions.length)*100;

  document.getElementById('q-counter').textContent = `Q${pracQIdx+1}/${pracQuestions.length}`;
  document.getElementById('q-diff-badge').textContent = q.difficulty;
  document.getElementById('q-diff-badge').style.cssText = `background:${dc}18;color:${dc};border:1px solid ${dc}30`;
  document.getElementById('q-marks-badge').textContent = `${q.marks} mark${q.marks>1?'s':''}`;
  document.getElementById('q-prog').style.width = prog + '%';

  document.getElementById('q-next-btn').style.display = 'none';

    <div class="option" id="opt-${i}" onclick="handleAnswer(${i})">
      <span class="opt-label" style="background:${dc}20;color:${dc}">${opt.label}</span>
      <span class="opt-text">${opt.text}</span>
    </div>`).join('');

  // Update question map
  document.getElementById('q-map').innerHTML = pracQuestions.map((q2, i) => {
    const a = pracAnswers[i];
    const dc2 = {Easy:'#00e5a0',Medium:'#ffb930',Hard:'#ff6b6b'}[q2.difficulty];
    let bg = i === pracQIdx ? '#a78bfa' : a !== undefined ? (a.selectedIndex===q2.correctIndex?'#00e5a0':'#ff6b6b') : dc2+'45';
    return `<div class="q-dot" style="background:${bg};border:1px solid ${i===pracQIdx?'#a78bfa':'transparent'}">${i+1}</div>`;
  }).join('');

  // Timer
  clearInterval(qTimerInterval); qElapsed = 0;
  document.getElementById('q-exp-time').textContent = `/${q.expectedTime}s`;
  document.getElementById('q-timer').textContent = `⏱ 0s`;
  document.getElementById('q-timer').style.color = 'var(--t2)';

    qElapsed++; totalElapsed++;
    document.getElementById('q-timer').textContent = `⏱ ${qElapsed}s`;

function handleAnswer(optIdx) {
  const q = pracQuestions[pracQIdx];
  if (pracAnswers[pracQIdx] !== undefined) return;
  clearInterval(qTimerInterval);
  pracAnswers[pracQIdx] = {selectedIndex: optIdx, timeTaken: qElapsed};

  // Show feedback
  q.options.forEach((opt, i) => {
    const el = document.getElementById('opt-' + i);
    el.classList.add('answered');
    if (opt.isCorrect) el.classList.add('correct');
    else if (i === optIdx) el.classList.add('wrong');
    const label = el.querySelector('.opt-label');
    if (opt.isCorrect) { label.textContent = '✓'; label.style.background='rgba(0,229,160,.2)'; label.style.color='#00e5a0'; }
    else if (i === optIdx) { label.textContent = '✗'; label.style.background='rgba(255,107,107,.15)'; label.style.color='#ff6b6b'; }
  });

  // Update map
  const dc = {Easy:'#00e5a0',Medium:'#ffb930',Hard:'#ff6b6b'}[q.difficulty];
  pracQuestions.forEach((q2, i) => {
    const a = pracAnswers[i];

function computeTestMetrics() {
  let correct=0,tt=0,ea=0,et=0,ma=0,mt=0,ha=0,ht=0,earned=0,totalMarks=0;
  pracQuestions.forEach((q,i) => {
    const a = pracAnswers[i]; if(!a) return;
    const ok = a.selectedIndex === q.correctIndex;
    tt += a.timeTaken || q.expectedTime; totalMarks += q.marks;
    if(ok){correct++; earned+=q.marks;}
    if(q.difficulty==='Easy'){et++; if(ok)ea++;}
    else if(q.difficulty==='Medium'){mt++; if(ok)ma++;}
    else{ht++; if(ok)ha++;}
  });
  const n=pracQuestions.length;
  const acc=n>0?(correct/n)*100:0;
  const avgT=n>0?tt/n:60;
  const expAvg=n>0?pracQuestions.reduce((s,q)=>s+q.expectedTime,0)/n:60;
  const spd=Math.max(0,Math.min(100,(expAvg/Math.max(avgT,1))*100));
  const perf=acc*.6+spd*.4;
  const mp=totalMarks>0?(earned/totalMarks)*100:0;
  return {accuracy:Math.round(acc),speedScore:Math.round(spd),performanceScore:Math.round(perf),
    correct,total:n,avgTime:Math.round(avgT),markPct:Math.round(mp),
    earned,totalMarks,passed:mp>=60,
    easyAcc:et>0?Math.round((ea/et)*100):0, medAcc:mt>0?Math.round((ma/mt)*100):0, hardAcc:ht>0?Math.round((ha/ht)*100):0};
}

function refreshReports() {
  const subjects = Object.keys(activeSyllabus);
  const totalCh = subjects.reduce((s,subj) => s+(activeSyllabus[subj].chapters?.length||0), 0);
  const doneCh = Object.values(chapterStats).filter(s => s.completed).length;
  const avgAcc = avg(allResults,'accuracy');
  const avgPerf = avg(allResults,'performanceScore');
  const n = allResults.length;

  document.getElementById('rep-kpis').innerHTML = [
    {l:'Completion',v:`${totalCh>0?Math.round((doneCh/totalCh)*100):0}%`,sub:`${doneCh}/${totalCh} chapters`,c:'#00e5a0'},
    {l:'Tests Done',v:n,sub:`${allResults.filter(r=>r.metrics.passed).length} passed`,c:'#ffb930'},
    {l:'Avg Accuracy',v:`${avgAcc}%`,sub:'all tests',c:'#a78bfa'},
    {l:'Mastery',v:avgPerf>=75?'Advanced':avgPerf>=50?'Growing':n>0?'Beginner':'—',sub:`${avgPerf}% score`,c:'#38bdf8'}
  ].map(s => `<div class="card" style="padding:16px">
    <div style="font-family:var(--fd);font-size:22px;font-weight:800;color:${s.c};line-height:1">${s.v}</div>
    <div style="font-size:12px;font-weight:600;color:var(--t1);margin-top:5px">${s.l}</div>
    <div style="font-size:11px;color:var(--t3);margin-top:2px">${s.sub}</div>
  </div>`).join('');

  // Chapter table
  const attempted = getAllChapters().filter(ch => chapterStats[ch.id]?.attempts > 0);
  if (attempted.length === 0) {
    document.getElementById('rep-ch-grid').innerHTML = `<p style="color:var(--t2);font-size:13px;grid-column:span 3">Complete tests to see chapter performance</p>`;
  } else {
    document.getElementById('rep-ch-grid').innerHTML = attempted.map(ch => {
      const st = chapterStats[ch.id];
      const color = getSubjColor(ch.subject);
      const sc = st.completed ? '#00e5a0' : st.bestScore>=50 ? '#ffb930' : '#ff6b6b';
      return `<div style="padding:12px 14px;background:var(--ink3);border-radius:11px;border:1px solid ${st.completed?color+'35':'var(--bd)'}">
        <div style="font-size:12px;font-weight:700;color:var(--t1);overflow:hidden;text-overflow:ellipsis;white-space:nowrap;margin-bottom:3px">${ch.name}</div>
        <div style="font-size:11px;color:var(--t3);margin-bottom:7px">${ch.subject.replace('_',' ')}</div>
        <div style="display:flex;justify-content:space-between;align-items:center;margin-bottom:5px">
          ${makeBadge(`${st.bestScore||0}%`, sc)}
          <span style="font-size:11px;color:var(--t3)">${st.attempts} att.</span>
        </div>
        ${makeBar(st.bestScore||0, sc, '', '', 4)}
        <span style="font-size:10px;color:${st.completed?'#00e5a0':'#ff6b6b'}">${st.completed?'✅ Completed':'⏳ In Progress'}</span>
      </div>`;
    }).join('');
  }
}

let activeReportBtn = null;
function generateReport(type, btn) {
  if (activeReportBtn) activeReportBtn.classList.remove('active');
  btn.classList.add('active');
  activeReportBtn = btn;

  const titles = {progress:'📈 Progress Report',subject:'📚 Subject Analysis',improvement:'🎯 Improvement Plan',parent:'👨‍👩‍👧 Parent Summary',exam:'🏆 Exam Strategy'};
  document.getElementById('rep-output-title').textContent = titles[type] || 'Report';

  // Simulate loading
  document.getElementById('rep-output').innerHTML = `<div class="loading-screen">
    <div class="spinner" style="width:32px;height:32px"></div>
    <p style="color:var(--t2);font-size:13px;margin-top:14px">Generating your AI report...</p>
  </div>`;

  setTimeout(() => {
    const content = MOCK_REPORTS[type] || 'Report content unavailable.';
    const studentName = student.name;
    const personalised = content.replace('Your child', `${studentName}`);
    document.getElementById('rep-output').innerHTML = `
      <div class="report-ai-badge"><span>🤖</span><span>AI-Generated · ${new Date().toLocaleDateString()}</span></div>
      <div class="report-text" id="report-text-content"></div>`;
    // Typing effect
    const el = document.getElementById('report-text-content');
    el.classList.add('typing-cursor');
    let ci = 0;
    const iv = setInterval(() => {
      if (ci < personalised.length) {
        el.textContent += personalised[ci++];
      } else {
        clearInterval(iv);
        el.classList.remove('typing-cursor');
      }
    }, 8);
  }, 1200);
}

let menuOpen = false;

function toggleUserMenu() {
  menuOpen = !menuOpen;
  const dd = document.getElementById('user-dropdown');
  const chevron = document.getElementById('umenu-chevron');
  dd.style.display = menuOpen ? 'block' : 'none';
  chevron.style.transform = menuOpen ? 'rotate(0deg)' : 'rotate(180deg)';
  if (menuOpen) {
    dd.style.animation = 'fadeUp .25s ease';
  }
}

// Close menu when clicking outside
document.addEventListener('click', function(e) {
  const btn = document.getElementById('user-menu-btn');
  const dd = document.getElementById('user-dropdown');
  if (btn && dd && !btn.contains(e.target) && !dd.contains(e.target)) {
    menuOpen = false;
    dd.style.display = 'none';
    const chevron = document.getElementById('umenu-chevron');
    if (chevron) chevron.style.transform = 'rotate(180deg)';
  }
});

function updateUserMenuDisplay() {
  if (!student) return;
  document.getElementById('umenu-avatar').textContent = student.avatar;
  document.getElementById('umenu-name').textContent = student.name;
}

const AVATARS_LIST = ["🦁","🐯","🦊","🐺","🦅","🐬","🌟","🚀","⚡","🔥","💎","🦋"];
let upSelectedAvatar = '';

function openUpdateProfile() {
  // Close dropdown
  menuOpen = false;
  document.getElementById('user-dropdown').style.display = 'none';

  // Pre-fill fields
  upSelectedAvatar = student.avatar;
  document.getElementById('up-name').value = student.name;
  document.getElementById('up-email').value = student.email || '';
  document.getElementById('up-board-display').textContent = student.board;
  document.getElementById('up-class-display').textContent = 'Class ' + student.cls;
  document.getElementById('up-success').style.display = 'none';
  document.getElementById('up-err-name').style.display = 'none';
  document.getElementById('up-err-email').style.display = 'none';

  // Build avatar grid
  document.getElementById('up-avatar-grid').innerHTML = AVATARS_LIST.map(a =>
    `<button onclick="upSelectAvatar(this,'${a}')"
      style="font-size:20px;padding:7px 9px;border-radius:9px;cursor:pointer;
        background:${a===upSelectedAvatar?'rgba(0,229,160,.1)':'#1c2030'};
        border:2px solid ${a===upSelectedAvatar?'var(--jade)':'#2c3148'};
        transition:all .2s">${a}</button>`
  ).join('');

  document.getElementById('modal-profile').classList.add('open');
}

function upSelectAvatar(btn, av) {
  upSelectedAvatar = av;
  document.querySelectorAll('#up-avatar-grid button').forEach(b => {
    b.style.background = '#1c2030';
    b.style.borderColor = '#2c3148';
  });
  btn.style.background = 'rgba(0,229,160,.1)';
  btn.style.borderColor = 'var(--jade)';
}

function saveProfile() {
  const name = document.getElementById('up-name').value.trim();
  const email = document.getElementById('up-email').value.trim();
  let ok = true;
  if (!name) { document.getElementById('up-err-name').style.display = 'block'; ok = false; }
  else document.getElementById('up-err-name').style.display = 'none';
  if (!email.includes('@')) { document.getElementById('up-err-email').style.display = 'block'; ok = false; }
  else document.getElementById('up-err-email').style.display = 'none';
  if (!ok) return;

  // Apply changes
  student.name = name;
  student.email = email;
  student.avatar = upSelectedAvatar;

  // Update all UI
  updateSidebar();
  updateUserMenuDisplay();
  renderDashboard();
  if (activePage === 'dashboard') renderDashboard();

  // Show success
  const succ = document.getElementById('up-success');
  succ.style.display = 'block';
  setTimeout(() => {
    succ.style.display = 'none';
    closeModal('modal-profile');
  }, 1800);
}

function openLogout() {
  menuOpen = false;
  document.getElementById('user-dropdown').style.display = 'none';

  // Fill logout modal with current student info
  document.getElementById('lo-avatar').textContent = student.avatar;
  document.getElementById('lo-name').textContent = student.name;

  const all = Object.values(chapterStats).length;
  const done = Object.values(chapterStats).filter(s => s.completed).length;
  const pct = all > 0 ? Math.round((done/all)*100) : 0;
  document.getElementById('lo-progress').textContent = pct + '%';

  document.getElementById('modal-logout').classList.add('open');
}

function confirmLogout() {
  closeModal('modal-logout');
  setTimeout(() => {
    document.getElementById('app').style.opacity = '0';
    document.getElementById('app').style.transition = 'opacity .4s ease';
    setTimeout(() => {
      document.getElementById('app').style.display = 'none';
      document.getElementById('app').style.opacity = '1';
      document.getElementById('app').style.transition = '';
      showLoginScreen();
    }, 420);
  }, 150);
}
